const frequenciesBy = (array, f) => {
  const ret = {};
  forEach(array, (item) => {
    const key = f(item);
    if (ret[key]) ret[key] += 1;
    else ret[key] = 1;
  });
  return ret;
};

const groupBy = (array, f) => {
  const ret = {};
  forEach(array, (item) => {
    const key = f(item);
    if (ret[key]) ret[key].push(item);
    else ret[key] = [item];
  });
  return ret;
};
